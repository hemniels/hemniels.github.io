#Umwandlung von y_train/y_test in onehot encoded format
y_train = to_categorical(y_train, num_classes=10)
y_test = to_categorical(y_test, num_classes=10)
input_shape = X_train.shape[1:]

#Erstellung unseres Classifier Models als Sequentielles Model
img_classifier = Sequential()

# Block 1
img_classifier.add(Conv2D(32, (3,3), padding='same', activation='relu', input_shape=input_shape))
img_classifier.add(Conv2D(32, (3,3), padding='same', activation='relu'))
img_classifier.add(MaxPooling2D(pool_size=(2,2)))
img_classifier.add(Dropout(0.25))

# Block 2
img_classifier.add(Conv2D(64, (3,3), padding='same', activation='relu'))
img_classifier.add(Conv2D(64, (3,3), padding='same', activation='relu'))
img_classifier.add(MaxPooling2D(pool_size=(2,2)))
img_classifier.add(Dropout(0.25))

# Block 3
img_classifier.add(Conv2D(128, (3,3), padding='same', activation='relu'))
img_classifier.add(Conv2D(128, (3,3), padding='same', activation='relu'))
img_classifier.add(MaxPooling2D(pool_size=(2,2)))
img_classifier.add(Dropout(0.25))

# Fully connected layers
img_classifier.add(Flatten())
img_classifier.add(Dense(512, activation='relu'))
img_classifier.add(Dropout(0.25))
img_classifier.add(Dense(10, activation='softmax'))


#Compile des Models
img_classifier.compile(loss='categorical_crossentropy',
                       optimizer='adam',
                       metrics=['accuracy'])

#Model fit
history = img_classifier.fit(X_train, y_train,
                             batch_size=64,
                             epochs=20,
                             validation_data=(X_test, y_test))

test_loss, test_acc = img_classifier.evaluate(X_test, y_test)
print('Test Genauigkeit:', test_acc)
